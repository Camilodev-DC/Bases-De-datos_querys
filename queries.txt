-- Query 1
SELECT
    t24.title AS FILM_title_2024, t24.views AS FILM_views_2024,
    t23.title AS FILM_title_2023, t23.views AS FILM_views_2023,
    t22.title AS TV_title_2022, t22.views AS TV_views_2024,
    t21.title AS TV_title_2021, t21.views AS TV_views_2023
FROM TV_netflix2024 AS t24
FULL JOIN TV_netflix20231 AS t23
ON t24.title = t23.title
FULL JOIN Fil_Netflix2024 AS t22
ON t24.title = t22.title OR t23.title = t22.title
FULL JOIN Film_netflix20231 AS t21
ON t24.title = t21.title OR t23.title = t21.title OR t22.title = t21.title;

-- Query 7
SELECT title
FROM (SELECT title, views
      FROM TV-netflix20231
      WHERE title LIKE 'ONE PIECE%'
      UNION ALL
      SELECT title, views
      FROM TV_netflix2024
      WHERE title LIKE 'Baby Reindeer%')
AS duelo
ORDER BY LENGTH(title) DESC, views DESC
LIMIT 1;

-- Query 13
SELECT
CASE substr(release_date, 6, 2)
    WHEN '01' THEN 'January'
    WHEN '02' THEN 'February'
    WHEN '03' THEN 'March'
    WHEN '04' THEN 'April'
    WHEN '05' THEN 'May'
    WHEN '06' THEN 'June'
    WHEN '07' THEN 'July'
    WHEN '08' THEN 'August'
    WHEN '09' THEN 'September'
    WHEN '10' THEN 'October'
    WHEN '11' THEN 'November'
    WHEN '12' THEN 'December'
END AS Mes,
Count(*) AS total
FROM Fil_netFlix 2024
GROUP BY substr(release_date, 6, 2)
ORDER BY total DESC
LIMIT 2;

-- Query 18
SELECT title,
SUM(CAST(REPLACE(Views, ',', '') AS DECIMAL)) AS Vistas_totales
FROM (SELECT Title, Views FROM TV_netflix20231
      UNION ALL
      SELECT Title, Views FROM TV-netflix2024
      UNION ALL
      SELECT title, views FROM Film-netflix20231
      UNION ALL
      SELECT title, views FROM Fil-netflix2024)
AS todas_las_vistas
GROUP BY title
ORDER BY Vistas_totales DESC;
